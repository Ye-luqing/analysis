\begin{exercise}
  若
$$
\left(\frac{b-a}{c-a}\right)=\left(\frac{a-c}{b-c}\right),
$$
$a,b,c,$ 成什么图形.
\end{exercise}
\begin{proof}[\textbf{解}]
是个等边三角形,如图.\\

\psline{->}(6.62,3.06)(8.48,-2.24)

\end{scriptsize}

\end{proof}
