$$
(a^2+b^2)(c^2+d^2)=|a+bi|^2|c+di|^2=|(a+bi)(c+di)|^2=|(ac-bd)+i(ad+bc)|^2=(ac-bd)^2+(ad+bc)^2.
$$
\end{proof}
